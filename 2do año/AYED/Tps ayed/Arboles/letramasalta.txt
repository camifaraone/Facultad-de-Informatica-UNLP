	public char letraMasAlta(ArbolGeneral<T> arbol){
		
		ColaGenerica<ArbolGeneral<T>> cola= new ColaGenerica<ArbolGeneral<T>>();
		cola.encolar(arbol);
		char max= this.getDatoRaiz();
		cola.encolar(null);
		ArbolGeneral<T> aux;
		ListaGenerica<ArbolGeneral<T>> hijos;
		while (!cola.esVacia()){
			aux= cola.desencolar();
			if (aux. getDatoRaiz() > max){
				max:=aux.getDatoRaiz();
			}
			if (aux != null){
				
				hijos= aux.getHijos();
				hijos.comenzar();
				while(!hijos.fin()){
					cola.encolar(hijos.proximo());
				}
			}else{ 				
				if (!cola.esVacia()){
					cola.encolar(null);
				}
				
			}
		}
		return max;
	}
